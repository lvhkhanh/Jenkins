## Samples

```
pipeline {
  agent any
  options {
    buildDiscarder(logRotator(numToKeepStr:'3'))
  }
  parameters {
    string(name: 'GREETING',
           defaultValue: 'Hello',
           description: 'The way to say hello')
  }
  stages {
    stage('Say Hi to ...') {
      steps {
        checkout scm:[
          $class:'GitSCM',
          branches:[[name:'*/master']],
          userRemoteConfigs:[[
            url:'file:///path/to/config']]]
      }
    }
  }
}
```
